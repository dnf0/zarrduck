//! Shared DuckDB session setup for eider tools (CLI and MCP), so extension
//! loading and the read sandbox can't drift between them.
use color_eyre::eyre::{Result as EyreResult, WrapErr};
use duckdb::Connection;

/// Load the eider loadable extension. Honors `EIDER_EXTENSION_PATH`, else
/// discovers `eider.duckdb_extension` next to the binary or under target/debug.
pub fn load_eider_extension(conn: &Connection) -> EyreResult<()> {
    let ext_name = "eider.duckdb_extension";
    if let Ok(explicit) = std::env::var("EIDER_EXTENSION_PATH") {
        if !explicit.is_empty() {
            return conn
                .execute(&format!("LOAD '{}'", explicit.replace('\'', "''")), [])
                .map(|_| ())
                .wrap_err_with(|| format!("Failed to load extension at {explicit}"));
        }
    }
    let mut candidates = vec![
        std::path::PathBuf::from(format!("./target/debug/{ext_name}")),
        std::path::PathBuf::from(format!("../target/debug/{ext_name}")),
    ];
    if let Ok(exe) = std::env::current_exe() {
        if let Some(p) = exe.parent() {
            candidates.push(p.join(ext_name));
            if let Some(gp) = p.parent() { candidates.push(gp.join(ext_name)); }
        }
    }
    let path = candidates.into_iter().find(|p| p.exists())
        .unwrap_or_else(|| std::path::PathBuf::from(format!("../target/debug/{ext_name}")));
    let s = path.to_string_lossy().replace('\'', "''");
    conn.execute(&format!("LOAD '{s}'"), [])
        .wrap_err_with(|| format!("Failed to load eider extension at {s}"))?;
    Ok(())
}

/// Open an in-memory DuckDB connection with the eider extension loaded.
pub fn open_connection() -> EyreResult<Connection> {
    let config = duckdb::Config::default()
        .allow_unsigned_extensions()
        .wrap_err("config unsigned extensions")?;
    let conn = Connection::open_in_memory_with_flags(config)
        .wrap_err("open in-memory duckdb")?;
    load_eider_extension(&conn).wrap_err("load eider extension")?;
    Ok(conn)
}

/// Open a session for the MCP/analytics use case: eider + the DuckDB `spatial`
/// extension loaded. `spatial` provides ST_Read / ST_Contains / ST_Intersects.
pub fn open_session() -> EyreResult<Connection> {
    let conn = open_connection()?;
    conn.execute_batch("INSTALL spatial; LOAD spatial;")
        .wrap_err("install/load spatial extension")?;
    Ok(conn)
}

use color_eyre::eyre::{Result as EyreResult, WrapErr};
use duckdb::Connection;
use serde_json::{json, Value};
use std::sync::atomic::{AtomicU64, Ordering};

static RESULT_SEQ: AtomicU64 = AtomicU64::new(0);

/// Materialize `select_sql` into a TEMP table, return {table_handle, row_count,
/// columns, rows (<=limit), truncated}. Full data stays in the temp table.
pub fn materialize(conn: &Connection, select_sql: &str, limit: usize) -> EyreResult<Value> {
    let n = RESULT_SEQ.fetch_add(1, Ordering::Relaxed);
    let handle = format!("mcp_result_{n}");
    conn.execute_batch(&format!("CREATE TEMP TABLE {handle} AS {select_sql}"))
        .wrap_err("materialize result")?;
    let row_count: i64 = conn.query_row(&format!("SELECT count(*) FROM {handle}"), [], |r| r.get(0))?;
    let rows = rows_as_json(conn, &format!("SELECT * FROM {handle} LIMIT {limit}"))?;
    let columns = column_names(conn, &handle)?;
    Ok(json!({
        "table_handle": handle,
        "row_count": row_count,
        "columns": columns,
        "rows": rows,
        "truncated": (row_count as usize) > limit,
    }))
}

/// Execute an arbitrary SELECT and return up to `limit` rows as JSON objects.
pub fn rows_as_json(conn: &Connection, sql: &str) -> EyreResult<Value> {
    let mut stmt = conn.prepare(sql).wrap_err("prepare query")?;
    let col_names: Vec<String> = {
        let q = stmt.query([])?;
        q.as_ref().map(|s| s.column_names()).unwrap_or_default()
    };
    // Re-run to iterate (query consumed above only for names if needed); simpler:
    let mut stmt = conn.prepare(sql)?;
    let mut rows = stmt.query([])?;
    let names = rows.as_ref().map(|s| s.column_names()).unwrap_or(col_names);
    let mut out = Vec::new();
    while let Some(row) = rows.next()? {
        let mut obj = serde_json::Map::new();
        for (i, name) in names.iter().enumerate() {
            obj.insert(name.clone(), duck_value_to_json(row, i));
        }
        out.push(Value::Object(obj));
    }
    Ok(Value::Array(out))
}

fn column_names(conn: &Connection, table: &str) -> EyreResult<Value> {
    let mut stmt = conn.prepare(&format!("SELECT * FROM {table} LIMIT 0"))?;
    let rows = stmt.query([])?;
    let names = rows.as_ref().map(|s| s.column_names()).unwrap_or_default();
    Ok(json!(names))
}

/// Best-effort DuckDB value -> JSON. Numbers as f64/i64, else string.
fn duck_value_to_json(row: &duckdb::Row, i: usize) -> Value {
    if let Ok(v) = row.get::<_, Option<f64>>(i) { return v.map(|x| json!(x)).unwrap_or(Value::Null); }
    if let Ok(v) = row.get::<_, Option<i64>>(i) { return v.map(|x| json!(x)).unwrap_or(Value::Null); }
    if let Ok(v) = row.get::<_, Option<String>>(i) { return v.map(|x| json!(x)).unwrap_or(Value::Null); }
    Value::Null
}

use crate::result::{materialize, rows_as_json};
use color_eyre::eyre::{eyre, Result as EyreResult};
use duckdb::Connection;
use serde_json::{json, Value};

const DEFAULT_LIMIT: usize = 1000;

fn esc(s: &str) -> String { s.replace('\'', "''") }

/// Build the optional bbox/time named args for read_geo from a JSON object.
fn pushdown_args(bbox: Option<&Value>, time: Option<&Value>, asset: Option<&str>) -> String {
    let mut parts = Vec::new();
    if let Some(b) = bbox {
        for (k, p) in [("lon_min","lon_min"),("lat_min","lat_min"),("lon_max","lon_max"),("lat_max","lat_max")] {
            if let Some(v) = b.get(k).and_then(|x| x.as_f64()) { parts.push(format!("{p} := {v}")); }
        }
    }
    if let Some(t) = time {
        if let Some(v) = t.get("min").and_then(|x| x.as_f64()) { parts.push(format!("time_min := {v}")); }
        if let Some(v) = t.get("max").and_then(|x| x.as_f64()) { parts.push(format!("time_max := {v}")); }
    }
    if let Some(a) = asset { parts.push(format!("asset := '{}'", esc(a))); }
    if parts.is_empty() { String::new() } else { format!(", {}", parts.join(", ")) }
}

pub fn describe_dataset(conn: &Connection, uri: &str) -> EyreResult<Value> {
    let sql = format!("SELECT * FROM read_zarr_metadata('{}')", esc(uri));
    Ok(json!({ "metadata": rows_as_json(conn, &sql)? }))
}

pub fn estimate_cost(conn: &Connection, uri: &str, bbox: Option<&Value>, time: Option<&Value>) -> EyreResult<Value> {
    let sql = format!("SELECT * FROM plan_read_geo('{}'{})", esc(uri), pushdown_args(bbox, time, None));
    Ok(json!({ "estimate": rows_as_json(conn, &sql)? }))
}

pub fn read_region(conn: &Connection, uri: &str, bbox: Option<&Value>, time: Option<&Value>, asset: Option<&str>, limit: Option<usize>) -> EyreResult<Value> {
    let sql = format!("SELECT * FROM read_geo('{}'{})", esc(uri), pushdown_args(bbox, time, asset));
    materialize(conn, &sql, limit.unwrap_or(DEFAULT_LIMIT))
}

pub fn list_tables(conn: &Connection) -> EyreResult<Value> {
    Ok(json!({ "tables": rows_as_json(conn, "SELECT table_name FROM information_schema.tables ORDER BY table_name")? }))
}

pub fn describe_table(conn: &Connection, name: &str) -> EyreResult<Value> {
    if !name.chars().all(|c| c.is_alphanumeric() || c == '_') {
        return Err(eyre!("invalid table name"));
    }
    Ok(json!({
        "schema": rows_as_json(conn, &format!("DESCRIBE {name}"))?,
        "sample": rows_as_json(conn, &format!("SELECT * FROM {name} LIMIT 5"))?,
    }))
}

```rust
/// Per-polygon metric over a grid asset. `convention`:
///  - "centroid"      -> ST_Contains(poly, cell_center)
///  - "all_touched"   -> ST_Intersects(poly, cell_box)
///  - "area_weighted" -> area-weighted (mean/sum only) via ST_Intersection
/// `metric` in {max,min,mean,sum,count}. `polygons` is a path read by ST_Read.
pub fn zonal_stats(conn: &Connection, grid_uri: &str, polygons: &str, metric: &str,
                   convention: &str, value_col: Option<&str>, limit: Option<usize>) -> EyreResult<Value> {
    let m = match metric { "max"|"min"|"mean"|"sum"|"count" => metric,
        _ => return Err(eyre!("metric must be one of max,min,mean,sum,count")) };
    let vc = value_col.unwrap_or("value");
    if !vc.chars().all(|c| c.is_alphanumeric() || c == '_') { return Err(eyre!("invalid value column")); }
    if convention == "area_weighted" && (m == "max" || m == "min") {
        return Err(eyre!("area_weighted applies to mean/sum, not max/min"));
    }
    // Push the polygons' combined bbox into read_geo so only intersecting chunks are read.
    let poly = esc(polygons);
    let setvar = format!(
        "SET VARIABLE _z_bbox = (SELECT ST_Extent_Agg(geom) FROM ST_Read('{poly}'));");
    let read = format!(
        "read_geo('{}', lon_min := ST_XMin(getvariable('_z_bbox')), lat_min := ST_YMin(getvariable('_z_bbox')), \
         lon_max := ST_XMax(getvariable('_z_bbox')), lat_max := ST_YMax(getvariable('_z_bbox')))", esc(grid_uri));
    conn.execute_batch(&setvar).wrap_err("compute polygons bbox")?;
    let select = match convention {
        "centroid" => format!(
            "SELECT v.*, {agg} AS metric FROM ST_Read('{poly}') v \
             JOIN {read} z ON ST_Contains(v.geom, ST_Point(z.lon, z.lat)) GROUP BY v.*",
            agg = agg_expr(m, "z", vc)),
        "all_touched" => format!(
            "SELECT v.*, {agg} AS metric FROM ST_Read('{poly}') v \
             JOIN {read} z ON ST_Intersects(v.geom, ST_MakeEnvelope(z.lon-0.0, z.lat-0.0, z.lon, z.lat)) GROUP BY v.*",
            agg = agg_expr(m, "z", vc)),
        "area_weighted" => format!(
            "SELECT v.*, sum(z.{vc}*ST_Area(ST_Intersection(v.geom, ST_MakeEnvelope(z.lon, z.lat, z.lon, z.lat)))) \
             / nullif(sum(ST_Area(ST_Intersection(v.geom, ST_MakeEnvelope(z.lon, z.lat, z.lon, z.lat)))),0) AS metric \
             FROM ST_Read('{poly}') v JOIN {read} z ON ST_Intersects(v.geom, ST_MakeEnvelope(z.lon, z.lat, z.lon, z.lat)) GROUP BY v.*"),
        _ => return Err(eyre!("convention must be centroid|all_touched|area_weighted")),
    };
    let mut out = materialize(conn, &select, limit.unwrap_or(DEFAULT_LIMIT))?;
    out["convention"] = json!(convention);
    out["metric"] = json!(metric);
    Ok(out)
}

fn agg_expr(metric: &str, alias: &str, col: &str) -> String {
    match metric {
        "count" => "count(*)".to_string(),
        _ => format!("{}({alias}.{col})", metric),
    }
}
```

The implementer must compute the half-cell offsets (as the zonal-stats benchmark does with `±d/2`) and build correct cell boxes; the placeholders above (`z.lon, z.lat`) MUST be replaced with the real cell extents. Verify the result matches `scripts/bench_polygons.sql` semantics. (`centroid` needs no cell box and is the simplest to get exactly right first; get it green, then all_touched/area_weighted.)

use color_eyre::eyre::{eyre, Result as EyreResult};

/// Allow read-only / temp-only statements; reject writes-to-disk, DDL on
/// persistent objects, attach, copy-to, and extension installs. Best-effort,
/// conservative: a single statement, first keyword allow-listed, no denied tokens.
pub fn ensure_read_only(sql: &str) -> EyreResult<()> {
    let trimmed = sql.trim().trim_end_matches(';');
    if trimmed.contains(';') {
        return Err(eyre!("only a single statement is allowed"));
    }
    let upper = trimmed.to_uppercase();
    let first = upper.split_whitespace().next().unwrap_or("");
    let allowed_first = ["SELECT", "WITH", "DESCRIBE", "EXPLAIN", "SHOW", "PRAGMA", "VALUES", "FROM", "TABLE", "SET", "CREATE"];
    if !allowed_first.contains(&first) {
        return Err(eyre!("statement type '{first}' is not permitted (read-only session)"));
    }
    // CREATE / SET are only allowed for TEMP tables / VARIABLE
    if first == "CREATE" && !(upper.contains("TEMP TABLE") || upper.contains("TEMPORARY TABLE") || upper.contains("TEMP VIEW") || upper.contains("TEMPORARY VIEW")) {
        return Err(eyre!("CREATE is only allowed for TEMP tables/views"));
    }
    if first == "SET" && !upper.starts_with("SET VARIABLE") {
        return Err(eyre!("only SET VARIABLE is allowed"));
    }
    for denied in ["INSTALL ", "ATTACH ", "COPY ", "EXPORT ", "INSERT ", "UPDATE ", "DELETE ", "DROP ", "ALTER ", "LOAD "] {
        if upper.contains(denied) {
            return Err(eyre!("token '{}' is not permitted", denied.trim()));
        }
    }
    Ok(())
}
